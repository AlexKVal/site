import { Spring } from 'react-motion'

view Circle {
  c = () => Math.round(Math.random() * 255)
  style = {
    background: `rgb(${c()}, ${c()}, ${c()})`,
    top: ^top, left: ^left,
    width: 40, height: 40,
    margin: [-20, 0, 0, -40],
    borderRadius: 100,
    position: 'absolute' }

  <Spring defaultValue={0} endValue={1}>
    {x => <circle style={{ ...style,
            transform: { scale: x }} } />}
  </Spring>
}
