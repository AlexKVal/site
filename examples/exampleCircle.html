import { Spring } from 'react-motion'

view Circle {
  c = () => Math.round(Math.random() * 255)
  base = {
    background: `rgb(${c()}, ${c()}, ${c()})`,
    top: ^top,
    left: ^left,
    width: 80, height: 80,
    marginLeft: -40,
    borderRadius: 100,
    position: 'absolute'
  }

  style = scale =>
    ({ ...base, transform: { scale } })

  <Spring defaultValue={0} endValue={1}>
    {x => <circle style={style(x)} />}
  </Spring>
}
