import { Spring } from 'react-motion'

view Circle {
  const c = () => Math.round(Math.random()*255)
  const base = {
    background: { r: c(), g: c(), b: c() },
    position: 'absolute',
    top: ^top,
    left: ^left,
    width: 80,
    height: 80,
    borderRadius: 100
  }

  const style = scale =>
    ({ ...base, transform: { scale } })

  <Spring endValue={{ val: 1 }}>{i =>
    <circle style={style(i.val)} />
  }</Spring>
}
